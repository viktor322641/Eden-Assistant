// ==UserScript==
// @name         Eden Assistant DOM Inspector
// @namespace    eden-assistant
// @version      0.5
// @description  Opens Eden 1 Vue and inspects real inputs, buttons, links and tabs without filling anything
// @match        https://login.eden1vision.com/*
// @match        https://eden.dealfile.co.uk/*
// @grant        none
// @run-at       document-idle
// ==/UserScript==

(function () {
    "use strict";

    const AUTO_HASH = "#eden-assistant-inspect";

    const sleep = milliseconds =>
        new Promise(resolve => setTimeout(resolve, milliseconds));

    function normalise(text) {
        return String(text || "")
            .replace(/\s+/g, " ")
            .trim()
            .toLowerCase();
    }

    function isVisible(element) {
        if (!element) return false;

        const style = window.getComputedStyle(element);
        const rect = element.getBoundingClientRect();

        return (
            style.display !== "none" &&
            style.visibility !== "hidden" &&
            rect.width > 0 &&
            rect.height > 0
        );
    }

    function setStatus(message, isError = false) {
        console.log("[Eden Assistant]", message);

        const status =
            document.getElementById("edenAssistantStatus");

        if (!status) return;

        status.textContent = message;
        status.style.background =
            isError ? "#b71c1c" : "#263238";
    }

    async function waitForElement(
        finder,
        timeoutMilliseconds = 20000
    ) {
        const started = Date.now();

        while (
            Date.now() - started < timeoutMilliseconds
        ) {
            const element = finder();

            if (element) return element;

            await sleep(400);
        }

        return null;
    }

    function findEdenVueLink() {
        const elements = [
            ...document.querySelectorAll(
                "a, div, span, button"
            )
        ];

        const textElement = elements.find(element => {
            return (
                isVisible(element) &&
                normalise(element.textContent) ===
                    "eden 1 vue"
            );
        });

        if (!textElement) return null;

        return textElement.closest("a") || textElement;
    }

    async function openEdenVue() {
        setStatus("Looking for Eden 1 Vue...");

        const link = await waitForElement(
            findEdenVueLink,
            10000
        );

        if (!link) {
            setStatus("Eden 1 Vue tile not found", true);
            return;
        }

        setStatus("Opening Eden 1 Vue...");

        if (link.tagName === "A" && link.href) {
            const targetUrl =
                link.href.split("#")[0] + AUTO_HASH;

            window.location.href = targetUrl;
            return;
        }

        link.click();
    }

    function safeText(value, maxLength = 180) {
        const text = String(value || "")
            .replace(/\s+/g, " ")
            .trim();

        if (text.length <= maxLength) return text;
        return text.slice(0, maxLength) + "…";
    }

    function describeElement(element, index) {
        const tag = element.tagName.toLowerCase();
        const visible = isVisible(element) ? "YES" : "NO";
        const rect = element.getBoundingClientRect();
        const label = element.labels?.length
            ? [...element.labels]
                .map(item => safeText(item.textContent, 80))
                .filter(Boolean)
                .join(" | ")
            : "";

        const nearbyText = safeText(
            element.closest("label, td, th, li, div, form")
                ?.innerText,
            180
        );

        const attributes = {
            type: element.getAttribute("type"),
            id: element.id,
            name: element.getAttribute("name"),
            class: element.className,
            placeholder: element.getAttribute("placeholder"),
            title: element.getAttribute("title"),
            ariaLabel: element.getAttribute("aria-label"),
            role: element.getAttribute("role"),
            value: element.value,
            href: element.href,
            onclick: element.getAttribute("onclick")
        };

        const lines = [
            `[${index}] <${tag}>`,
            `visible: ${visible}`,
            `position: x=${Math.round(rect.x)}, y=${Math.round(rect.y)}, width=${Math.round(rect.width)}, height=${Math.round(rect.height)}`
        ];

        for (const [key, value] of Object.entries(attributes)) {
            const clean = safeText(value, 220);
            if (clean) lines.push(`${key}: ${clean}`);
        }

        const ownText = safeText(
            element.innerText || element.textContent,
            180
        );

        if (ownText) lines.push(`text: ${ownText}`);
        if (label) lines.push(`label: ${label}`);
        if (nearbyText && nearbyText !== ownText) {
            lines.push(`nearby: ${nearbyText}`);
        }

        return lines.join("\n");
    }

    function collectElements() {
        const selectors = [
            "input",
            "button",
            "select",
            "textarea",
            "a",
            "[role='button']",
            "[role='tab']",
            "input[type='submit']",
            "input[type='button']"
        ].join(",");

        return [...new Set(
            document.querySelectorAll(selectors)
        )];
    }

    function createReport() {
        const elements = collectElements();
        const pageTitle = document.title || "";

        const header = [
            "EDEN ASSISTANT DOM REPORT",
            `Generated: ${new Date().toISOString()}`,
            `URL: ${location.href}`,
            `Title: ${pageTitle}`,
            `Total elements: ${elements.length}`,
            "",
            "IMPORTANT: This report only reads the page. It does not fill or click WIP/Search/Inspection.",
            ""
        ].join("\n");

        const body = elements
            .map((element, index) =>
                describeElement(element, index + 1)
            )
            .join("\n\n------------------------------\n\n");

        return header + body;
    }

    function closeInspector() {
        document.getElementById("edenDomInspectorOverlay")?.remove();
    }

    function showInspector(report) {
        closeInspector();

        const overlay = document.createElement("div");
        overlay.id = "edenDomInspectorOverlay";

        Object.assign(overlay.style, {
            position: "fixed",
            inset: "0",
            zIndex: "2147483647",
            background: "rgba(0,0,0,.88)",
            display: "flex",
            flexDirection: "column",
            padding: "12px",
            boxSizing: "border-box"
        });

        const toolbar = document.createElement("div");

        Object.assign(toolbar.style, {
            display: "flex",
            gap: "8px",
            marginBottom: "8px",
            flexWrap: "wrap"
        });

        const copyButton = document.createElement("button");
        copyButton.textContent = "COPY REPORT";

        const closeButton = document.createElement("button");
        closeButton.textContent = "CLOSE";

        for (const button of [copyButton, closeButton]) {
            Object.assign(button.style, {
                padding: "11px 14px",
                border: "1px solid white",
                borderRadius: "8px",
                background: "#1565c0",
                color: "white",
                fontWeight: "bold"
            });
        }

        closeButton.style.background = "#616161";

        const textarea = document.createElement("textarea");
        textarea.value = report;
        textarea.readOnly = true;

        Object.assign(textarea.style, {
            flex: "1",
            width: "100%",
            boxSizing: "border-box",
            resize: "none",
            padding: "12px",
            borderRadius: "8px",
            background: "#111",
            color: "#fff",
            fontFamily: "monospace",
            fontSize: "12px",
            lineHeight: "1.45"
        });

        copyButton.addEventListener("click", async () => {
            try {
                await navigator.clipboard.writeText(report);
                copyButton.textContent = "COPIED ✓";
                setStatus("DOM report copied");
            } catch (error) {
                textarea.focus();
                textarea.select();
                document.execCommand("copy");
                copyButton.textContent = "COPIED ✓";
                setStatus("DOM report copied");
            }
        });

        closeButton.addEventListener("click", closeInspector);

        toolbar.appendChild(copyButton);
        toolbar.appendChild(closeButton);
        overlay.appendChild(toolbar);
        overlay.appendChild(textarea);
        document.body.appendChild(overlay);
    }

    async function inspectPage() {
        setStatus("Scanning page elements...");

        await sleep(500);

        const report = createReport();
        showInspector(report);

        const total = collectElements().length;
        setStatus(`DOM report ready: ${total} elements`);
    }

    async function runAssistant() {
        const button =
            document.getElementById("edenAssistantButton");

        if (button) {
            button.disabled = true;
            button.textContent = "WORKING...";
        }

        try {
            if (
                location.hostname === "login.eden1vision.com"
            ) {
                await openEdenVue();
                return;
            }

            if (
                location.hostname === "eden.dealfile.co.uk"
            ) {
                await inspectPage();
                return;
            }

            setStatus("Unsupported page", true);
        } catch (error) {
            console.error(error);
            setStatus(`Error: ${error.message || error}`, true);
        } finally {
            if (button) {
                button.disabled = false;
                button.textContent =
                    location.hostname === "login.eden1vision.com"
                        ? "OPEN EDEN 1 VUE"
                        : "SCAN PAGE";
            }
        }
    }

    function createPanel() {
        if (
            document.getElementById("edenAssistantPanel")
        ) {
            return;
        }

        if (!document.body) {
            setTimeout(createPanel, 500);
            return;
        }

        const panel = document.createElement("div");
        panel.id = "edenAssistantPanel";

        Object.assign(panel.style, {
            position: "fixed",
            right: "10px",
            bottom: "85px",
            zIndex: "2147483646",
            display: "flex",
            flexDirection: "column",
            alignItems: "flex-end",
            gap: "8px"
        });

        const status = document.createElement("div");
        status.id = "edenAssistantStatus";
        status.textContent = "DOM Inspector ready";

        Object.assign(status.style, {
            maxWidth: "290px",
            padding: "9px 12px",
            borderRadius: "9px",
            background: "#263238",
            color: "#ffffff",
            fontSize: "13px",
            boxShadow: "0 3px 10px rgba(0,0,0,.45)"
        });

        const button = document.createElement("button");
        button.id = "edenAssistantButton";
        button.textContent =
            location.hostname === "login.eden1vision.com"
                ? "OPEN EDEN 1 VUE"
                : "SCAN PAGE";

        Object.assign(button.style, {
            padding: "14px 17px",
            border: "2px solid white",
            borderRadius: "12px",
            background: "#1565c0",
            color: "#ffffff",
            fontSize: "15px",
            fontWeight: "bold",
            boxShadow: "0 3px 10px rgba(0,0,0,.45)"
        });

        button.addEventListener("click", runAssistant);

        panel.appendChild(status);
        panel.appendChild(button);
        document.body.appendChild(panel);
    }

    createPanel();

    new MutationObserver(createPanel).observe(
        document.documentElement,
        {
            childList: true,
            subtree: true
        }
    );

    if (
        location.hostname === "eden.dealfile.co.uk" &&
        location.hash === AUTO_HASH
    ) {
        history.replaceState(
            null,
            "",
            location.pathname + location.search
        );

        setTimeout(inspectPage, 1500);
    }
})();
